import java.util.Scanner;

public class Main {
  public static void main(String[] args) {
    
    Scanner sc = new Scanner(System.in);
    int size = sc.nextInt();
    
    int[] arr = new int[size];

    for(int i=0; i<size; i++){
      arr[i] = sc.nextInt();
    }
    
    int min = 999999999;

    for(int i=0; i<size; i++){
      if(min>arr[i]){
        min = arr[i];
      }
    }

    System.out.println(min);
  }
}
